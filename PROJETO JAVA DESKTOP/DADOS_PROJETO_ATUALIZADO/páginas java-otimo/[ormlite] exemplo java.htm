83 private void readWriteData() throws Exception {
84 // create an instance of Account
85 String name = "Jim Coakley";
86 Account account = new Account(name);
87
88 // persist the account object to the database
89 accountDao.create(account);
90 int id = account.getId();
91 verifyDb(id, account);
92
93 // assign a password
94 account.setPassword("_secret");
95 // update the database after changing the object
96 accountDao.update(account);
97 verifyDb(id, account);
98
99 // query for all items in the database
100 List<Account> accounts = accountDao.queryForAll();
101 assertEquals("Should have found 1 account matching our query", 1, accounts.size());
102 verifyAccount(account, accounts.get(0));
103
104 // loop through items in the database
105 int accountC = 0;
106 for (Account account2 : accountDao) {
107 verifyAccount(account, account2);
108 accountC++;
109 }
110 assertEquals("Should have found 1 account in for loop", 1, accountC);
111
112 // construct a query using the QueryBuilder
113 QueryBuilder<Account, Integer> statementBuilder = accountDao.queryBuilder();
114 // shouldn't find anything: name LIKE 'hello" does not match our account
115 statementBuilder.where().like(Account.NAME_FIELD_NAME, "hello");
116 accounts = accountDao.query(statementBuilder.prepare());
117 assertEquals("Should not have found any accounts matching our query", 0, accounts.size());
118
119 // should find our account: name LIKE 'Jim%' should match our account
120 statementBuilder.where().like(Account.NAME_FIELD_NAME, name.substring(0, 3) + "%");
121 accounts = accountDao.query(statementBuilder.prepare());
122 assertEquals("Should have found 1 account matching our query", 1, accounts.size());
123 verifyAccount(account, accounts.get(0));
124
125 // delete the account since we are done with it
126 accountDao.delete(account);
127 // we shouldn't find it now
128 assertNull("account was deleted, shouldn't find any", accountDao.queryForId(id));
129 }

134 private void readWriteBunch() throws Exception {
135
136 Map<String, Account> accounts = new HashMap<String, Account>();
137 for (int i = 1; i <= 100; i++) {
138 String name = Integer.toString(i);
139 Account account = new Account(name);
140 // persist the account object to the database, it should return 1
141 accountDao.create(account);
142 accounts.put(name, account);
143 }
144
145 // query for all items in the database
146 List<Account> all = accountDao.queryForAll();
147 assertEquals("Should have found same number of accounts in map", accounts.size(), all.size());
148 for (Account account : all) {
149 assertTrue("Should have found account in map", accounts.containsValue(account));
150 verifyAccount(accounts.get(account.getName()), account);
151 }
152
153 // loop through items in the database
154 int accountC = 0;
155 for (Account account : accountDao) {
156 assertTrue("Should have found account in map", accounts.containsValue(account));
157 verifyAccount(accounts.get(account.getName()), account);
158 accountC++;
159 }
160 assertEquals("Should have found the right number of accounts in for loop", accounts.size(), accountC);
161 }

163 /**
164 * Example of created a query with a ? argument using the {@link SelectArg} object. You then can set the value of
165 * this object at a later time.
166 */
167 private void useSelectArgFeature() throws Exception {
168
169 String name1 = "foo";
170 String name2 = "bar";
171 String name3 = "baz";
172 assertEquals(1, accountDao.create(new Account(name1)));
173 assertEquals(1, accountDao.create(new Account(name2)));
174 assertEquals(1, accountDao.create(new Account(name3)));
175
176 QueryBuilder<Account, Integer> statementBuilder = accountDao.queryBuilder();
177 SelectArg selectArg = new SelectArg();
178 // build a query with the WHERE clause set to 'name = ?'
179 statementBuilder.where().like(Account.NAME_FIELD_NAME, selectArg);
180 PreparedQuery<Account> preparedQuery = statementBuilder.prepare();
181
182 // now we can set the select arg (?) and run the query
183 selectArg.setValue(name1);
184 List<Account> results = accountDao.query(preparedQuery);
185 assertEquals("Should have found 1 account matching our query", 1, results.size());
186 assertEquals(name1, results.get(0).getName());
187
188 selectArg.setValue(name2);
189 results = accountDao.query(preparedQuery);
190 assertEquals("Should have found 1 account matching our query", 1, results.size());
191 assertEquals(name2, results.get(0).getName());
192
193 selectArg.setValue(name3);
194 results = accountDao.query(preparedQuery);
195 assertEquals("Should have found 1 account matching our query", 1, results.size());
196 assertEquals(name3, results.get(0).getName());
197 }

203 private void useTransactions(ConnectionSource connectionSource) throws Exception {
204 String name = "trans1";
205 final Account account = new Account(name);
206 assertEquals(1, accountDao.create(account));
207
208 TransactionManager transactionManager = new TransactionManager(connectionSource);
209 try {
210 // try something in a transaction
211 transactionManager.callInTransaction(new Callable<Void>() {
212 public Void call() throws Exception {
213 // we do the delete
214 assertEquals(1, accountDao.delete(account));
215 assertNull(accountDao.queryForId(account.getId()));
216 // but then (as an example) we throw an exception which rolls back the delete
217 throw new Exception("We throw to roll back!!");
218 }
219 });
220 fail("This should have thrown");
221 } catch (SQLException e) {
222 // expected
223 }
224
225 assertNotNull(accountDao.queryForId(account.getId()));
226 }
